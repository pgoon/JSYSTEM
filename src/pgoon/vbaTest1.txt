Public hoshoNo As String
Public r As Integer

Private Sub kensakuBtn_Click()

    Dim c As Range
    
    Dim maxCellCnt As Integer
    
    maxCellCnt = Worksheets("火災契約情報").Range("B5").End(xlDown).Row

Label1:
    If hoshoNo <> hoshoBango.Text Then
      Set c = Range("B4:B" & maxCellCnt).Find(hoshoBango.Text)
       If Not c Is Nothing Then
        c.Select
        r = c.Row + 1
        hoshoNo = hoshoBango.Text
       End If
        
    Else
     
      Set c = Range("B" & r & ":B" & maxCellCnt).Find(hoshoBango.Text)
      
      If Not c Is Nothing Then
         c.Select
         r = c.Row + 1
         hoshoNo = hoshoBango.Text
      Else
         hoshoNo = ""
         GoTo Label1
      End If
         
    End If
     
End Sub
